18class MovingVoter: 

19 """ 

20 This subclass of `Embeddings` can be used to see 

21 what happen to the scores of the different candidates 

22 when a voter moves from a group to another. 

23 

24 There is 4 candidates and 3 groups: 

25 Each group strongly support one of 

26 the candidate and dislike the other 

27 candidates, except the last candidate 

28 which is fine for every group. 

29 

30 The moving voter is a voter that do not have any 

31 preference between the candidates 

32 (she gives a score of `0.8` to every candidate, 

33 except `0.5` for the last one), but her embeddings 

34 move from one position to another. 

35 

36 Parameters 

37 ---------- 

38 embeddings: Embeddings 

39 The embeddings of the voters. If none is specified, embeddings are the identity matrix. 

40 moving_voter: int 

41 The index of the voter that is moving 

42 

43 

44 Attributes 

45 ---------- 

46 rule: Rule 

47 The rule we are using in the election. 

48 moving_voter: int 

49 The index of the voter that is moving. 

50 ratings_: np.ndarray 

51 The ratings given by the voters to the candidates 

52 

53 

54 Examples 

55 -------- 

56 >>> moving_profile = MovingVoter() 

57 >>> moving_profile(RuleSumRatings()) # DOCTEST: +ELLIPSIS 

58 <embedded_voting.experiments.moving_voter.MovingVoter object at ...> 

59 >>> moving_profile.moving_voter 

60 0 

61 >>> moving_profile.embeddings 

62 Embeddings([[1., 0., 0.], 

63 [0., 0., 1.], 

64 [0., 1., 0.], 

65 [1., 0., 0.]]) 

66 >>> moving_profile.ratings_ 

67 Ratings([[0.8, 0.8, 0.8, 0.5], 

68 [0.1, 0.1, 1. , 0.5], 

69 [0.1, 1. , 0.1, 0.5], 

70 [1. , 0.1, 0.1, 0.5]]) 

71 """ 

72 def __init__(self, embeddings=None, moving_voter=0): 

73 self.rule = None 

74 if embeddings is None: 

75 embeddings = np.array([[1., 0., 0.], [0., 0., 1.], [0., 1., 0.], [1., 0., 0.]]) 

76 self.embeddings = Embeddings(embeddings, norm=True) 

77 self.moving_voter = moving_voter 

78 self.ratings_ = None 

79 

80 def __call__(self, rule, ratings=None): 

81 """ 

82 This function is used to update the rule used, and also the ratings 

83 

84 Parameters 

85 ---------- 

86 rule: Rule 

87 The rule we will use 

88 ratings: np.ndarray or Ratings 

89 The ratings of the candidates 

90 

91 Return 

92 ------ 

93 MovingVoter 

94 The object 

95 """ 

96 self.rule = rule 

97 if ratings is None: 

98 ratings = np.array([[.8, .8, .8, .5], [.1, .1, 1, .5], [.1, 1, .1, .5], [1, .1, .1, .5]]) 

99 self.ratings_ = Ratings(ratings) 

100 return self

102 def plot_scores_evolution(self, show=True): 

103 """ 

104 This function plot the evolution 

105 of the scores of the candidates 

106 when the moving voters' embeddings 

107 are changing. 

108 

109 Parameters 

110 ---------- 

111 show : bool 

112 If True, displays the figure 

113 at the end of the function. 

114 

115 Examples 

116 -------- 

117 >>> p = MovingVoter()(RuleSVDNash()) 

118 >>> p.plot_scores_evolution(show=False) 

119 """ 

120 

121 tab_x = np.linspace(0, 1, 50) 

122 tab_y = [] 

123 for x in tab_x: 

124 self.embeddings[self.moving_voter] = normalize([1-x, x, 0]) 

125 tab_y.append(self.rule(self.ratings_, self.embeddings).scores_focus_on_last_) 

126 

127 tab_y = np.array(tab_y).T 

128 name = ["Start", "End", "Orthogonal", "Consensus"] 

129 for i in range(self.ratings_.n_candidates): 

130 plt.plot(tab_x, tab_y[i], label=name[i]) 

131 

132 plt.title("Evolution of the score") 

133 plt.xlabel("X coordinate of moving voter") 

134 plt.ylabel("Score") 

135 plt.xlim(0, 1) 

136 plt.legend() 

137 if show: 

138 plt.show() # pragma: no cover

140 def plot_features_evolution(self, show=True): 

141 """ 

142 This function plot the evolution 

143 of the features of the candidates 

144 when the moving voters' embeddings 

145 are changing. Only works for :class:`SVDMax` and :class:`RuleFeatures`. 

146 

147 Parameters 

148 ---------- 

149 show : bool 

150 If True, displays the figure 

151 at the end of the function. 

152 

153 Examples 

154 -------- 

155 >>> p = MovingVoter()(RuleSVDMax()) 

156 >>> p.plot_features_evolution(show=False) 

157 """ 

158 tab_x = np.linspace(0, 1, 20) 

159 tab_y = [] 

160 for x in tab_x: 

161 self.embeddings[self.moving_voter] = normalize([1-x, x, 0]) 

162 tab_y.append(self.rule(self.ratings_, self.embeddings).features_) 

163 tab_y = np.array(tab_y) 

164 

165 fig = plt.figure(figsize=(10, 5)) 

166 

167 ax = create_3d_plot(fig, position=[1, 2, 1]) 

168 name = ["Start", "End", "Orthogonal", "Consensus"] 

169 for i in range(self.ratings_.n_candidates): 

170 vec_init = normalize(tab_y[0, i])**2 

171 ax.plot(tab_y[::, i, 0], tab_y[::, i, 1], tab_y[::, i, 2], 

172 color=(vec_init[0] * 0.8, vec_init[1] * 0.8, vec_init[2] * 0.8), 

173 alpha=0.5, label=name[i]) 

174 for j, v in enumerate(tab_y[::, i]): 

175 vec_normalized = normalize(v) 

176 ax.plot([0, v[0]], [0, v[1]], [0, v[2]], 

177 color=(vec_normalized[0] * 0.8, vec_normalized[1] * 0.8, vec_normalized[2] * 0.8), 

178 alpha=j / 60) 

179 ax.set_title("Evolution of the features") 

180 plt.legend() 

181 

182 tax = create_ternary_plot(fig, position=[1, 2, 2]) 

183 for i in range(self.ratings_.n_candidates): 

184 points = [normalize(x[[0, 2, 1]])**2 for x in tab_y[::, i]] 

185 vec_init = normalize(tab_y[0, i, [0, 2, 1]])**2 

186 tax.plot(points, color=(vec_init[0] * 0.8, vec_init[2] * 0.8, vec_init[1] * 0.8), alpha=0.8) 

187 tax.scatter([vec_init], 

188 color=(vec_init[0] * 0.8, vec_init[2] * 0.8, vec_init[1] * 0.8), 

189 alpha=0.7, s=50) 

190 

191 if show: 

192 plt.show() # pragma: no cover
